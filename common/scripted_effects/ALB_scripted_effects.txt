ALB_increase_italian_influence = {
	custom_effect_tooltip = r56_newline
	if = {
		limit = {
			has_idea = ALB_idea_italian_influence_1
		}
		swap_ideas = {
			remove_idea = ALB_idea_italian_influence_1
			add_idea = ALB_idea_italian_influence_2
		}
	}
	else_if = {
		limit = {
			has_idea = ALB_idea_italian_influence_2
		}
		swap_ideas = {
			remove_idea = ALB_idea_italian_influence_2
			add_idea = ALB_idea_italian_influence_3
		}
	}
	else_if = {
		limit = {
			has_idea = ALB_idea_italian_influence_3
		}
		swap_ideas = {
			remove_idea = ALB_idea_italian_influence_3
			add_idea = ALB_idea_italian_influence_4
		}
	}
	else = {
		effect_tooltip = {
			add_ideas = ALB_idea_italian_influence_4
		}
	}
}
