		}

		on_map_mode = map_only

		days_remove = 90
		
		cost = 50

        state_target = yes

		visible = {
			original_tag = ALB

				has_idea = ALB_idea_illiteracy_2
				has_idea = ALB_idea_illiteracy_3
			}
		}

        target_trigger = {
            FROM = {
				NOT = {
					has_state_flag = ALB_school_opened

		modifier = {

		}

		complete_effect = {
			set_country_flag = {

				days = 90
			}
		}

        remove_effect = {
			FROM = {
                set_state_flag = ALB_funding_school_flag
            }
			clr_country_flag = ALB_funding_school_flag
			if = {
				limit = {
					has_idea = ALB_idea_illiteracy_1
				}
				swap_ideas = {
					remove_idea = ALB_idea_illiteracy_1
					add_idea = ALB_idea_illiteracy_2
				}
			}
			else_if = {
				limit = {
					has_idea = ALB_idea_illiteracy_2
				}
				swap_ideas = {
					remove_idea = ALB_idea_illiteracy_2
					add_idea = ALB_idea_illiteracy_3
				}
			}
			else_if = {
				limit = {
					has_idea = ALB_idea_illiteracy_3
				}
					remove_ideas = ALB_idea_illiteracy_3
			}
			set_country_flag = {
				flag = ALB_funding_school_flag_timer
				value = 1
				days = 60
			}
        }

        ai_will_do = {
            base = 20
        } 
	}
